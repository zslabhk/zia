function FindProxyForURL(url, host) {
var drop = "BLOCK";
/* Return DIRECT to Allow access */
if ((localHostOrDomainIs(host, "hkjc.com")) ||
(localHostOrDomainIs(host, "hk01.com")) ||
(localHostOrDomainIs(host, "google.com")) ||
(localHostOrDomainIs(host, "zscaler.com")) )
return "DIRECT";
/* Default Block Statement to block anything not allowed above */
return drop;
}
